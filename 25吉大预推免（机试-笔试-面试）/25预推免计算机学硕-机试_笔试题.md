字符串中提取数字，求被7整除的数的个数

```cpp
#include <bits/stdc++.h>

using namespace std;
using LL = long long;

#define int LL
#define endl "\n"

signed main() {
	cin.tie(nullptr)->sync_with_stdio(false);
	string s;
	getline(cin, s);
	
	int x = 0, cnt = 0;
	for (int i = 0; i < s.size(); i++) {
		if ('0' <= s[i] && s[i] <= '9') {
			x = x * 10 + s[i] - '0';
		} else {
			if (x > 0 && x % 7 == 0) {
				cnt += 1;
			}
			x = 0;
		}
	}
	cout << cnt << endl;
	return 0;
}
```

求集合的子集（二进制枚举）

 ```
#include <bits/stdc++.h>

using namespace std;
using LL = long long;

#define int LL
#define endl "\n"

signed main() {
	cin.tie(nullptr)->sync_with_stdio(false);
	int n;
	cin >> n;
	
	string s;
	cin >> s;
	
	for (int mask = 0; mask < 1 << n; mask++) {
		for (int i = 0; i < n; i++) {
			if (mask >> i & 1) {
				cout << s[i];
			}
		}
		cout << endl;
	}
	return 0 ^ 0;
}
 ```
